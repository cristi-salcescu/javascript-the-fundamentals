function createRateIt(){
  const ratings = [];
  
  return function(rating){
    if(rating){
      ratings.push(rating);
    }
    
    const total = ratings.reduce((x, y) => x + y);
    return total / ratings.length;
  }
}

const rateIt = createRateIt();
rateIt(5);
rateIt(3);
rateIt(4);

const avgRating = rateIt();
